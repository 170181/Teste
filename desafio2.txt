SELECT B.NOME AS 'DEPARTAMENTO',
	   A.NOME AS 'PESSOA',
	   MAX(A.SALARIO) AS 'SALARIO'
  	   FROM PESSOA A
       INNER JOIN DEPARTAMENTO B ON A.DEPTID=B.ID
	   WHERE A.DEPTID = 1

UNION

SELECT B.NOME AS 'DEPARTAMENTO',
	   A.NOME AS 'PESSOA',
	   MAX(A.SALARIO) AS 'SALARIO'
  	   FROM PESSOA A
       INNER JOIN DEPARTAMENTO B ON A.DEPTID=B.ID
	   WHERE A.DEPTID = 2
